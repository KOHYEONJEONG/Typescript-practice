# 슈퍼 타입 -- x --> 서브 타입
# 슈퍼 타입 <-- o -- 서브 타입

let num1: number = 10;
let num2: 10 = 10;
num2 = num1 ; //에러(다운캐스트 불가)

num1 = num2; //가능(업 캐스트) 


let num1 : 10  = 10; // number literal Type
let num2 : number  = 10;

num1 = num2;//에러 (다운 캐스팅)
num2 = num1;


즉. 
⭕ 업캐스팅=> 부모타입 = 자식타입(= : 대입연산자)
❌ 다운캐스팅=>  자식타입 = 부모타입 



Q. TypeScript의 타입 호환성 규칙에 따라, 상위 타입-하위 타입 관계에서 보통 안전하게 허용되는 값 할당은 어떤 방식일까요?
A. ⭕하위 타입 값을 상위 타입 변수에 할당
=> 하위 타입 값을 상위 타입으로 취급하는 것을 업캐스팅이라고 하며, 이는 대부분의 경우 안전하게 허용됩니다. 반대인 다운캐스팅은 일반적으로 허용되지 않아요.
